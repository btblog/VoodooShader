00028     {
00029         const wchar_t * formatBody = format;
00030

00032         {
00033             formatBody += 3;

00035 
00036         if (formatBody[0] == VSTR('D'))

00038             // Depth format
00039             if (_tcsnicmp(formatBody, VSTR("D16"), 3) == 0) return VSFmt_D16;

00049             if (_tcsnicmp(formatBody, VSTR("RGBA32F"), 7) == 0) return VSFmt_RGBA32F;
00050         }
00051
